### 2.2 채팅 위젯 임베드 시스템

고객사 웹사이트에 단 한 줄의 코드로 채팅 위젯을 설치할 수 있습니다:

```html
<script src="https://cdn.aihelp.kr/widget.js" data-key="YOUR_API_KEY"></script>
```

위젯 특징:
- **경량**: gzip 기준 < 30KB
- **프레임워크 무관**: 바닐라 JS로 구현, Shadow DOM으로 스타일 격리
- **반응형**: 모바일/데스크톱 자동 대응
- **커스터마이징**: 색상, 위치, 언어 등 설정 가능
- **오프라인 모드**: 에이전트 불가 시 이메일 폴백

### 2.3 실시간 통신

```typescript
// WebSocket 연결 관리 (자동 재연결 포함)
class ChatWebSocket {
  private ws: WebSocket;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;

  connect(conversationId: string) {
    this.ws = new WebSocket(
      `wss://api.aihelp.kr/ws/chat/${conversationId}`
    );

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      switch (data.type) {
        case 'ai_response':      // AI 응답
        case 'typing':           // 타이핑 인디케이터
        case 'agent_transfer':   // 상담원 전환 알림
        case 'satisfaction':     // 만족도 조사
      }
    };
  }
}
```

### 4.2 문서 처리 파이프라인

```python
class DocumentProcessor:
    """문서 업로드부터 벡터 인덱싱까지의 파이프라인"""

    SUPPORTED_FORMATS = ['pdf', 'docx', 'csv', 'txt', 'html', 'md']

    async def process(self, document: UploadedDocument) -> int:
        # 1단계: 텍스트 추출
        raw_text = await self.extract_text(document)

        # 2단계: 전처리 (클리닝, 정규화)
        cleaned_text = self.preprocess(raw_text)

        # 3단계: 청크 분할 (의미 기반)
        chunks = self.split_into_chunks(
            cleaned_text,
            chunk_size=512,           # 토큰 기준
            chunk_overlap=50,         # 문맥 유지를 위한 오버랩
            strategy="semantic"       # 의미 기반 분할 (vs. fixed-size)
        )

        # 4단계: 임베딩 생성 + 벡터 DB 저장
        embeddings = await self.embedding_service.batch_embed(chunks)
        await self.vector_store.upsert(
            documents=chunks,
            embeddings=embeddings,
            metadata=[{
                "doc_id": document.id,
                "org_id": document.org_id,
                "source": document.source_type,
                "chunk_index": i
            } for i in range(len(chunks))]
        )

        return len(chunks)
```

### 4.3 검색 및 응답 생성

```python
class RAGService:
    """질문에 대한 관련 문서 검색 및 응답 생성"""

    async def answer(
        self,
        query: str,
        org_id: str,
        conversation_history: list[Message]
    ) -> RAGResponse:

        # 1. 쿼리 임베딩 생성
        query_embedding = await self.embedding_service.embed(query)

        # 2. 벡터 유사도 검색 (해당 조직의 문서만)
        relevant_chunks = await self.vector_store.similarity_search(
            query_embedding=query_embedding,
            filter={"org_id": org_id},
            top_k=5,
            score_threshold=0.7       # 최소 유사도 임계값
        )

        # 3. 리랭킹 (Cross-encoder로 정밀도 향상)
        reranked = await self.reranker.rerank(query, relevant_chunks)

        # 4. 프롬프트 구성
        prompt = self.build_prompt(
            query=query,
            context=reranked[:3],     # 상위 3개 청크 사용
            history=conversation_history[-5:],  # 최근 5개 대화
            system_prompt=self.get_system_prompt(org_id)
        )

        # 5. LLM 응답 생성 (스트리밍)
        response = await self.llm.generate(
            prompt=prompt,
            model="gpt-4o",
            temperature=0.3,          # 사실성 중시
            max_tokens=500,
            stream=True
        )

        # 6. 할루시네이션 검증
        confidence = await self.verify_response(
            response=response,
            sources=reranked[:3]
        )

        return RAGResponse(
            answer=response,
            sources=reranked[:3],
            confidence=confidence
        )
```

### 4.4 LLM 비용 최적화 전략

| 전략 | 설명 | 예상 절감 |
|------|------|-----------|
| **모델 캐스케이드** | 간단한 질문 → GPT-4o-mini, 복잡한 질문 → GPT-4o | ~40% |
| **응답 캐싱** | 동일/유사 질문에 대한 Redis 캐시 활용 | ~25% |
| **프롬프트 최적화** | 토큰 수 최소화, 효율적 컨텍스트 구성 | ~15% |
| **배치 임베딩** | 문서 임베딩 시 배치 처리로 API 호출 최소화 | ~20% |

### 5.2 오케스트레이터 구현

```python
class AgentOrchestrator:
    """LangGraph 기반 멀티에이전트 오케스트레이터"""

    def __init__(self):
        self.intent_classifier = IntentClassifier()
        self.agents = {
            "faq": FAQAgent(),
            "order": OrderAgent(),
            "escalation": EscalationAgent()
        }
        self.graph = self._build_graph()

    def _build_graph(self) -> StateGraph:
        """LangGraph 상태 그래프 구성"""
        graph = StateGraph(ConversationState)

        # 노드 정의
        graph.add_node("classify", self.classify_intent)
        graph.add_node("faq", self.agents["faq"].process)
        graph.add_node("order", self.agents["order"].process)
        graph.add_node("escalation", self.agents["escalation"].process)
        graph.add_node("respond", self.compose_response)

        # 엣지 정의 (조건부 라우팅)
        graph.add_edge(START, "classify")
        graph.add_conditional_edges(
            "classify",
            self.route_to_agent,
            {
                "faq": "faq",
                "order": "order",
                "escalation": "escalation"
            }
        )
        graph.add_edge("faq", "respond")
        graph.add_edge("order", "respond")
        graph.add_edge("escalation", "respond")
        graph.add_edge("respond", END)

        return graph.compile()

    async def classify_intent(self, state: ConversationState) -> dict:
        """사용자 의도를 분류하여 적절한 에이전트로 라우팅"""
        intent = await self.intent_classifier.classify(
            message=state["current_message"],
            history=state["conversation_history"],
            org_context=state["org_settings"]
        )
        return {
            "intent": intent.category,        # faq | order | escalation
            "confidence": intent.confidence,   # 0.0 ~ 1.0
            "entities": intent.entities        # 추출된 엔티티
        }

    def route_to_agent(self, state: ConversationState) -> str:
        """의도에 따른 에이전트 라우팅 로직"""
        # 낮은 신뢰도 → 에스컬레이션
        if state["confidence"] < 0.5:
            return "escalation"

        # 감정 분석 결과 부정적 → 에스컬레이션
        if state.get("sentiment_score", 0) < -0.5:
            return "escalation"

        return state["intent"]
```

```python
class EscalationAgent:
    """상담원 연결 및 에스컬레이션 관리 에이전트"""

    async def process(self, state: ConversationState) -> dict:
        # 감정 분석
        sentiment = await self.analyze_sentiment(state["current_message"])

        # 에스컬레이션 사유 판단
        reason = self.determine_reason(state)

        # 우선순위 산정
        priority = self.calculate_priority(
            sentiment_score=sentiment.score,
            conversation_length=len(state["conversation_history"]),
            customer_tier=state.get("customer_tier", "standard"),
            topic_severity=state.get("topic_severity", "normal")
        )

        # 상담원에게 전달할 컨텍스트 요약 생성
        summary = await self.generate_handoff_summary(state)

        # 대기 시간 예측
        estimated_wait = await self.predict_wait_time(
            org_id=state["org_id"],
            priority=priority
        )

        return {
            "response": f"전문 상담원에게 연결해 드리겠습니다. "
                       f"예상 대기 시간은 약 {estimated_wait}분입니다. "
                       f"잠시만 기다려 주세요.",
            "escalation": {
                "priority": priority,      # critical, high, normal, low
                "reason": reason,
                "summary": summary,
                "sentiment": sentiment.score,
                "ticket_created": True
            }
        }
```

```python
class AISecurityLayer:
    """AI 파이프라인 보안 레이어"""

    async def sanitize_input(self, message: str) -> str:
        """입력 데이터 보안 처리"""
        # 1. 프롬프트 인젝션 탐지 및 차단
        if await self.detect_prompt_injection(message):
            raise SecurityException("잠재적 프롬프트 인젝션 탐지")

        # 2. PII 탐지 및 마스킹
        message = await self.mask_pii(message)
        # "010-1234-5678" → "010-****-****"
        # "hong@email.com" → "h***@email.com"

        return message

    async def sanitize_output(self, response: str) -> str:
        """출력 데이터 보안 처리"""
        # 1. 내부 시스템 정보 누출 방지
        response = self.remove_system_info(response)

        # 2. 민감한 비즈니스 정보 필터링
        response = await self.filter_sensitive_content(response)

        return response
```

## 부록: 기술 의사결정 기록 (ADR)

### ADR-001: FastAPI 선택 (vs. Django, Express)

**결정**: FastAPI를 주 백엔드 프레임워크로 선택

**근거**:
- Python 기반으로 AI/ML 라이브러리(LangChain, OpenAI SDK) 호환 우수
- 비동기(async/await) 네이티브 지원으로 높은 동시성 처리
- Pydantic 기반 자동 검증 및 OpenAPI 문서 자동 생성
- Django 대비 가볍고, Express 대비 Python 생태계 활용 가능

### ADR-002: ChromaDB 시작 → Pinecone 전환 계획

**결정**: MVP에서는 ChromaDB 사용, 1,000+ 고객 도달 시 Pinecone 전환

**근거**:
- ChromaDB: 무료, 로컬 실행 가능, 빠른 프로토타이핑
- Pinecone: 관리형 서비스, 자동 스케일링, 고가용성
- 전환 비용: 임베딩 재생성 필요하나, 추상화 레이어로 최소화

### ADR-003: 멀티에이전트 vs. 단일 LLM 체인

**결정**: LangGraph 기반 멀티에이전트 아키텍처 채택

**근거**:
- 각 에이전트가 특화된 도구와 프롬프트를 가질 수 있음
- 새로운 기능 추가 시 기존 에이전트에 영향 없음 (느슨한 결합)
- 에이전트별 독립적인 테스트 및 성능 최적화 가능
- 향후 고객사별 커스텀 에이전트 추가 용이
